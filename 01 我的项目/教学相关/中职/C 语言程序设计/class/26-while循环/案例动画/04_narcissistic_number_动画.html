练习4：水仙花数（各位立方和=自身）

代码

int num=100, count=0;
while (num <= 999) {
  int ge=num%10;
  int shi=num/10%10;
  int bai=num/100;
  int sum=ge*ge*ge+shi*shi*shi+bai*bai*bai;
  if (sum == num) {
    printf("%d\\n", num);
    count++;
  }
  num++;
}
