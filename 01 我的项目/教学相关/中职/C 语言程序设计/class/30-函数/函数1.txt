三、程序改错题（20 分）
主函数 main() 调用 fun() 函数，其中 fun 函数的功能是：找出 100 到 999 之间（含 100 和 999）所有各位上数字之和为 x（x 为正整数）
的整数并输出，符合条件的整数个数作为函数值返回。例如，当 x 值为 5 时，
满足条件的整数有：104, 113, 122, 131, 140, 203, 212, 220, 230, 302, 311, 320, 401, 410, 500，一共 15 个。
#include<stdio.h>
void fun(int x)
{
    int n,s1,s2,s3,t;
    n=0;
    t=100;
    while(t<=999)
    {
        s1=t%10; s2=t%10%10; s3=t/100;
        if(s1+s2+s3==x)
        {
            printf("%5d",t);
            n++;
        }
        t++;
    }
    return n;
}
int main()
{
    int x=-1,y;
    while(x<=0)
    {
        printf("Please input x(x>0):");
        scanf("%d",&x);
    }
    fun(x);
    printf("\nThe result is: %d\n",y);
}
