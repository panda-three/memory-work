main() {
    char s[80];
    int i=0;
    gets(s);
    while(s[i] != '\0') {
        if(s[i] <= 'z' && s[i] >= 'a')
            s[i] = s[i] - 'a' + 'A';
        i++;
    }
    puts(s);
}
