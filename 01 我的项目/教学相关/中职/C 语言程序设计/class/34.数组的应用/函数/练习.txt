}

2.#include <stdio.h>
void swap1(int c[])
{
    int t;
    t = c[0]; c[0] = c[1]; c[1] = t;
}
void swap2(int c0, int c1)
{
    int t;
    t = c0; c0 = c1; c1 = t;
}
void main()
{
    int a[2] = {4, 7}, b[2] = {4, 7};
    swap1(a);
    swap2(b[0], b[1]);
    printf("%d, %d, %d, %d\n", a[0],a[1], b[0], b[1]);
}

3.#include <stdio.h>
func(int a[][3])
{
   int i, j, sum = 0;
   for(i = 0; i < 3; i++)
       for(j = 0; j < 3; j++)
       {
           a[i][j] = i + j;
           if(i == j) sum = sum + a[i][j];
       }
   return sum;
}
void main()
{
   int a[3][3] = {1, 3, 5, 7, 9, 11, 13, 15, 17};
   int sum;
   sum = func(a);
   printf("sum = %d", sum);
}
