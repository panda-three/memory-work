顺序比较法排序

最直观的排序方法，通过逐个比较找出最小值

算法原理

顺序比较法是最简单的排序算法。它通过两两比较数组中的元素， 将较小的元素向前移动，经过多轮比较后，数组就会变得有序。

  1. 从第一个元素开始，依次与后面的所有元素比较
  2. 如果发现比当前元素小的，就交换位置
  3. 每轮比较后，当前位置都会是剩余元素中的最小值
  4. 重复这个过程直到整个数组有序
C语言实现 
// 顺序比较法排序
void sequentialSort(int arr[], int n) {
    int i, j, temp;
    
    // 外层循环：确定每个位置的元素
    for (i = 0; i < n - 1; i++) {
        // 内层循环：将第i个元素与后面的所有元素比较
        for (j = i + 1; j < n; j++) {
            // 如果后面的元素更小，则交换
            if (arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}

// 打印数组
void printArray(int arr[], int n) {
    int i;
    for (i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

// 使用示例
int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);
    
    printf("排序前: ");
    printArray(arr, n);
    
    sequentialSort(arr, n);
    
    printf("排序后: ");
    printArray(arr, n);
    
    return 0;
}

选择法排序

每次选择最小值，放到正确位置

算法原理

选择排序将数组分为已排序和未排序两部分。每次从未排序部分选择最小值， 放到已排序部分的末尾。重复这个过程直到整个数组有序。

  1. 在未排序部分找到最小元素的索引
  2. 将最小元素与未排序部分的第一个元素交换
  3. 扩展已排序部分的边界
  4. 重复直到所有元素都排序完成
C语言实现 
// 选择法排序
void selectionSort(int arr[], int n) {
    int i, j, min_idx, temp;
    
    // 逐个位置放置元素
    for (i = 0; i < n - 1; i++) {
        // 假设当前位置是最小的
        min_idx = i;
        
        // 在未排序部分寻找真正的最小值
        for (j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }
        
        // 将最小值交换到正确位置
        if (min_idx != i) {
            temp = arr[i];
            arr[i] = arr[min_idx];
            arr[min_idx] = temp;
        }
    }
}

// 优化版：减少不必要的交换
void selectionSortOptimized(int arr[], int n) {
    int i, j, min_idx, temp;
    
    for (i = 0; i < n - 1; i++) {
        min_idx = i;
        for (j = i + 1; j < n; j++) {
            if (arr[j] < arr[min_idx]) {
                min_idx = j;
            }
        }
        
        // 只有当找到更小的元素时才交换
        if (min_idx != i) {
            temp = arr[i];
            arr[i] = arr[min_idx];
            arr[min_idx] = temp;
        }
    }
}

冒泡法排序

相邻元素比较，大数"冒泡"到后面

算法原理

冒泡排序通过重复遍历数组，比较相邻元素并交换位置。 每次遍历都会将当前未排序部分的最大值"冒泡"到末尾。

  1. 从数组开始比较相邻的两个元素
  2. 如果前一个比后一个大，交换它们的位置
  3. 继续向后比较，直到数组末尾
  4. 重复遍历，每次减少比较范围
C语言实现 
// 冒泡法排序（基础版）
void bubbleSort(int arr[], int n) {
    int i, j, temp;
    
    // 外层循环控制排序轮数
    for (i = 0; i < n - 1; i++) {
        // 内层循环进行相邻元素比较
        for (j = 0; j < n - i - 1; j++) {
            // 如果前一个元素大于后一个，则交换
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

// 优化版：增加标志位
void bubbleSortOptimized(int arr[], int n) {
    int i, j, temp;
    int swapped;  // 标志位，记录是否发生交换
    
    for (i = 0; i < n - 1; i++) {
        swapped = 0;
        for (j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // 交换元素
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = 1;  // 标记发生了交换
            }
        }
        
        // 如果没有发生交换，说明数组已经有序
        if (swapped == 0) {
            break;
        }
    }
}

// 双向冒泡（鸡尾酒排序）
void cocktailSort(int arr[], int n) {
    int swapped = 1;
    int start = 0;
    int end = n - 1;
    int i, temp;
    
    while (swapped) {
        swapped = 0;
        
        // 从左到右冒泡
        for (i = start; i < end; i++) {
            if (arr[i] > arr[i + 1]) {
                temp = arr[i];
                arr[i] = arr[i + 1];
                arr[i + 1] = temp;
                swapped = 1;
            }
        }
        
        if (!swapped) break;
        
        swapped = 0;
        end--;
        
        // 从右到左冒泡
        for (i = end - 1; i >= start; i--) {
            if (arr[i] > arr[i + 1]) {
                temp = arr[i];
                arr[i] = arr[i + 1];
                arr[i + 1] = temp;
                swapped = 1;
            }
        }
        
        start++;
    }
}

插入法排序

将元素插入到已排序部分的正确位置

算法原理

插入排序将数组分为已排序和未排序两部分。每次从未排序部分取出第一个元素， 插入到已排序部分的正确位置。类似于整理扑克牌的过程。

  1. 将第一个元素视为已排序序列
  2. 取出下一个元素，在已排序序列中从后向前扫描
  3. 找到合适位置并插入元素
  4. 重复直到所有元素都插入完成
C语言实现 
// 插入法排序（基础版）
void insertionSort(int arr[], int n) {
    int i, j, key;
    
    // 从第二个元素开始
    for (i = 1; i < n; i++) {
        key = arr[i];  // 待插入的元素
        j = i - 1;
        
        // 将大于key的元素向后移动
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        
        // 插入key到正确位置
        arr[j + 1] = key;
    }
}

// 二分插入排序（优化版）
void binaryInsertionSort(int arr[], int n) {
    int i, j, key, left, right, mid;
    
    for (i = 1; i < n; i++) {
        key = arr[i];
        left = 0;
        right = i - 1;
        
        // 二分查找插入位置
        while (left <= right) {
            mid = (left + right) / 2;
            if (arr[mid] > key) {
                right = mid - 1;
            } else {
                left = mid + 1;
            }
        }
        
        // 移动元素并插入
        for (j = i - 1; j >= left; j--) {
            arr[j + 1] = arr[j];
        }
        arr[left] = key;
    }
}

// 希尔排序（插入排序的改进版）
void shellSort(int arr[], int n) {
    int gap, i, j, temp;
    
    // 从大间隔开始，逐渐减小间隔
    for (gap = n / 2; gap > 0; gap /= 2) {
        // 对每个间隔进行插入排序
        for (i = gap; i < n; i++) {
            temp = arr[i];
            for (j = i; j >= gap && arr[j - gap] > temp; j -= gap) {
                arr[j] = arr[j - gap];
            }
            arr[j] = temp;
        }
    }
}
