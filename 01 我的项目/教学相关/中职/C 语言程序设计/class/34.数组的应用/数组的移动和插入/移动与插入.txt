#include <stdio.h>
void main()
{
    char s[80]="abcdefgh", c;
    int i, n, k;
    n=0;
    for(i=0; s[i]!='\0'; i++) 
        n++;
    i=0;
    /************* FOUND(1) **********/
    if(n%2==1)
    	k=n-1;
    else
        k=n-2;
    c=s[k];
    for(i=k-2; i>=1; i=i-2)
        /************ FOUND(2) ********/
    	s[i+1]=s[i];
    /************ FOUND(3) ********/
    s[0]=c;
    puts(s);
}
