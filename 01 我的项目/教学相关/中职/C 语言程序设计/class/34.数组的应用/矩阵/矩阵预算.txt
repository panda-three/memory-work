二、程序阅读题（共 5 分）
#include <stdio.h>
void main()
{
    int i,j;
    int a[3][3] = {1,2,3,4,5,6,7,8,9};
    int b[2][2] = {11,22,33,44};
    int c[3][3];
    for(i=0;i<3;i++)
        for(j=0;j<3;j++)
            if (i>=2||j>=2)
                c[i][j]=a[i][j];
            else
                c[i][j]=a[i][j]+b[i][j];
    for(i=0;i<3;i++)
    {
        for(j=0;j<3;j++)
            printf("%d\t",c[i][j]);
        printf("\n");
    }
}

1. #include <stdio.h>
void main()
{
    int i,j;
    float v;
    float a[4][3] = {11,2,13,14,-5,16,17,18,-9,-2,-23,24};
    for(i=0;i<4;i++)
    {
        v=a[i][0];
        for(j=1;j<3;j++)
            v+=a[i][j];
        v/=3;
        for(j=0;j<3;j++)
            a[i][j] -= v;
    }
    for(i=0;i<4;i++)
    {
        for(j=0;j<3;j++)
            printf("%.1f\t",a[i][j]);
        printf("\n");
    }
}

2. #include <stdio.h>
#define N 3
void main()
{
    int a[N][N],b[N*N]={1,1},i,j;
    for(i=2;i<N*N;i++)
        b[i] = b[i-1]+b[i-2];
    for(i=0;i<N;i++)
        for(j=0;j<N;j++)
            a[j][i] = b[i*N+j];
    for(i=0;i<N;i++)
    {
        for(j=0;j<N;j++)
        printf("%d\t",a[i][j]);
     printf("\n");
    }
}
