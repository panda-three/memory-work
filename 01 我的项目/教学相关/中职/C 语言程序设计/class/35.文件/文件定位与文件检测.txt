#include<stdio.h>
void main()
{
    int a[4]={1,2,3,4},b;
    FILE *fp;
    fp=fopen("data.dat","wb");  // 二进制写模式
    for(i=0;i<4;i++)
        fwrite(&a[i],sizeof(int),1,fp);  // 逐个写入数组元素
    fclose(fp);

    fp=fopen("data.dat","rb");  // 二进制读模式
    fseek(fp,-1*sizeof(int),SEEK_END);  // 指针移到文件末尾向前1个int的位置
    fread(&b,sizeof(int),1,fp);  // 读取该位置的int值到b
    fclose(fp);
    printf("%d\n",b);
}

#include<stdio.h>
#include<stdlib.h>  // exit()需要该头文件
void main()
{
    FILE *fp;
    fp=fopen("a.txt","r");
    if(fp==NULL)
    {
        printf("cannot open a.txt\n");
        exit(0);
    }
    // 第一遍读取并输出
    while(!feof(fp))
        putchar(fgetc(fp));
    rewind(fp);  // 文件指针回到文件开头
    // 第二遍读取并输出
    while(!feof(fp))
        putchar(fgetc(fp));
    fclose(fp);
}


#include<stdio.h>
#include<stdlib.h>
#include<string.h>
void main()
{
    FILE *fp;
    int flag;
    char string[20],data[20];
    if((fp=fopen( (1) "bc.txt", "r+" ))==NULL)  // 打开目标文件
    {
        printf("open file error");
        exit(0);
    }
    do
    {
        printf("enter a string:");
        gets(string);
        if(strlen(string)==0)  // 输入空字符串，退出循环
            break;
        strcat(string,"\n");  // 拼接换行符，保持文件中字符串格式统一
        (2) rewind(fp);       // 重置文件指针到开头，确保每次查找从首行开始
        flag=1;               // 标记：1=未找到，0=已找到
        while(flag && (3) fgets(data,20,fp)!=NULL)  // 循环读取文件中的字符串
        {
            if(strcmp(data,string)==0)  // 对比读取的字符串与输入字符串
                flag=0;       // 找到匹配项，修改标记
        }
        if(flag)  // 未找到时，写入文件
            (4) fputs(string, fp);  // 将输入字符串写入文件
        else  // 找到时，提示信息
            printf("the string exists!\n");
    }while(1);  // 无限循环，直到输入空字符串退出
    fclose(fp);  // 关闭文件
}


#include<stdio.h>
void main()
{
    FILE *fp1, *fp2;
    char ch;
    fp1 = fopen( (1) "text.txt", "r" );  // 打开源文件（只读模式）
    fp2 = fopen("text1.txt","w");        // 打开目标文件（写入模式）
    while(!feof(fp1))                    // 循环读取，直到源文件结束
    {
        ch=fgetc(fp1);                   // 从源文件读取一个字符
        putchar(ch);                     // 输出到屏幕
        (4) fputc(ch, fp2);              // 将字符写入目标文件
    }
    fclose(fp1);                         // 关闭源文件
    fclose(fp2);                         // 关闭目标文件
}
